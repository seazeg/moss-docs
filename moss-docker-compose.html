通过 docker-compose 部署 ​

通过 docker-compose 负责对所有依赖容器集群的快速编排

加载镜像 tar

shell
docker load -i moss-server-lastest.tar

新建对应 Dockerfile

  • moss-server
Dockerfile
FROM moss-server:lastest
  • mysql

TIP

init.sql 文件

Dockerfile
FROM mysql:5.7
ENV AUTO_RUN_DIR /docker-entrypoint-initdb.d
ENV INSTALL_DB_SQL init.sql
COPY ./$INSTALL_DB_SQL $AUTO_RUN_DIR/
RUN chmod 777 $AUTO_RUN_DIR/$INSTALL_DB_SQL
  • redis
Dockerfile
FROM redis

新建 docker-compose.yml

yml
version: "3.0"

services:
    redis:
        container_name: redis
        build:
            context: ./docker/redis/
            dockerfile: Dockerfile
        ports:
            - "6379:6379"
        restart: on-failure
        networks:
            - bridge

    mysql:
        container_name: mysql
        build:
            context: ./docker/mysql/
            dockerfile: Dockerfile
        ports:
            - "3307:3306"
        restart: on-failure
        environment:
            - MYSQL_ROOT_PASSWORD=zhou.
        volumes:
            - /docker/moss/mysql/mysql-files:/var/lib/mysql-files
            - /docker/moss/mysql/conf:/etc/mysql/conf.d
            - /docker/moss/mysql/data:/var/lib/mysql
            - /docker/moss/mysql/log:/var/log/mysql
        networks:
            - bridge

    server:
        container_name: moss-server
        build:
            context: ./docker/moss/
            dockerfile: Dockerfile
        deploy:
            mode: replicated
            replicas: 1
            placement:
                constraints:
                    - node.labels.moss-server == true
            resources:
                limits:
                    cpus: "4"
                    memory: 8G
                reservations:
                    cpus: "2"
                    memory: 4G
        ports:
            - "7002:7002"
        restart: on-failure
        volumes:
            - /docker/moss/server/logs:/usr/moss/app/logs
        networks:
            - bridge
        depends_on:
            - redis
            - mysql

networks:
    bridge:

启动 docker-compose

shell
docker-compose -f docker-compose.yml up -d
